CREATE DATABASE camstore;

CREATE TABLE PRODUCTS (
  id SERIAL NOT NULL Primary key,
  name VARCHAR(50) NOT NULL,
  price NUMERIC NOT NULL,
  NOTES VARCHAR(100),
  creationDate TIMESTAMP NOT NULL);
